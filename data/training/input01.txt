		os << indentStr << "<" << __name() << " version=\"" << __ver() << "\">" << std::endl;
		
		// 让基类先处理

		}
		
		// 输出XML尾部
		os << indentStr << "</" << __name() << ">" << std::endl;

	{

		if (bRepaired)
			*bRepaired = FALSE;
			
		// 让基类先处理
		GObjBase *base = _base();
		if (base)
			base->LoadXml(xml, bRepaired);
			
		// 处理所有元素
		T *owner = _owner();
		GElemBase *p = __elems();
		while(p)
		{
			if (!p->LoadXml(owner, xml))
			{
				if (bRepaired)
					*bRepaired = TRUE;
			}
			p = p->next;
		}
		
		return TRUE;
	}
	
	virtual void SaveXmlToFile(const std::string &filename)
	{
		std::ofstream file(filename.c_str());
		if (file.is_open())
		{
			file << "<?xml version=\"1.0\" encoding=\"UTF-8\"?>" << std::endl;
			SaveXml(file);
			file.close();
		}
	}
	
	virtual BOOL LoadXmlFromFile(const std::string &filename, BOOL *bRepaired = NULL)
	{
		std::ifstream file(filename.c_str());
		if (!file.is_open())
			return FALSE;
			
		std::stringstream buffer;
		buffer << file.rdbuf();
		file.close();
		
		return LoadXml(buffer.str(), bRepaired);
	}
