	virtual void LoadSimple(CDataPacket &dp)
	{
		T *owner=_owner();
		GElemBase *p=__elems();
		while(p)
		{
			p->Load(owner,dp,TRUE);
			p=p->next;
		}
	}

	virtual void SaveXml(pugi::xml_node &parent_node)
	{
		pugi::xml_node class_node = parent_node.append_child(__name().c_str());
		class_node.append_attribute("ver") = (unsigned int)__ver();
		if (__uid() >= GOBJ_UID_START)
		{
			class_node.append_attribute("uid") = __uid();
		}

		// Save base class if it exists
		GObjBase *base = _base();
		if (base)
		{
			base->SaveXml(class_node);
		}

		T *owner = _owner();
		GElemBase *elem = __elems();
		while (elem)
		{
			elem->SaveXml(owner, class_node);
			elem = elem->next;
		}
	}

	virtual BOOL LoadXml(const pugi::xml_node &node)
	{
		if (strcmp(node.name(), __name().c_str()) != 0)
		{
			// TODO: Add proper error handling or logging
			return FALSE; 
		}

		pugi::xml_attribute ver_attr = node.attribute("ver");
		if (!ver_attr || ver_attr.as_uint() != __ver())
		{
			// TODO: Add proper error handling or logging for version mismatch
			return FALSE;
		}

		pugi::xml_attribute uid_attr = node.attribute("uid");
		if (__uid() >= GOBJ_UID_START)
		{
			if (!uid_attr || uid_attr.as_uint() != __uid())
			{
				// TODO: Add proper error handling or logging for UID mismatch
				return FALSE;
			}
		}
		
		// Load base class if it exists
		GObjBase *base = _base();
		if (base)
		{
			// Assuming the base class is the first child node if it's saved that way.
			// This might need adjustment based on how base class XML is structured.
			pugi::xml_node base_node = node.child(base->GetName());
			if (base_node)
			{
				if (!base->LoadXml(base_node))
				{
					return FALSE; // Failed to load base
				}
			}
			// Else: Base class node not found, could be an error or optional
		}


		T *owner = _owner();
		std::unordered_map<std::string, GElemBase*> elem_map;
		for (GElemBase *el = __elems(); el; el = el->next)
		{
			elem_map[el->GetElemName()] = el;
		}

		for (pugi::xml_node elem_node = node.child(nullptr); elem_node; elem_node = elem_node.next_sibling(nullptr))
		{
			// Skip if this node is for the base class (already processed or not this class's element)
			if (base && strcmp(elem_node.name(), base->GetName()) == 0)
			{
				continue;
			}

			auto it = elem_map.find(elem_node.name());
			if (it != elem_map.end())
			{
				GElemBase *elem = it->second;
				// Check version of element if stored
				pugi::xml_attribute elem_ver_attr = elem_node.attribute("ver");
				if (elem_ver_attr && elem_ver_attr.as_uint() != elem->GetVer())
				{
					// Version mismatch for element, skip or error
					// TODO: Add proper error handling
					continue;
				}
				if (!elem->LoadXml(owner, elem_node))
				{
					// Failed to load element
					// TODO: Add proper error handling
					return FALSE; 
				}
			}
			// else: Element in XML not found in class definition, or an unknown child.
			// This could be an error or skipped if forward compatibility is desired.
		}
		return TRUE;
	}

		vector_node.append_attribute("type").set_value(subtype.c_str()); // Class name of T
		vector_node.append_attribute("ver").set_value(ver);
		vector_node.append_attribute("elem_type").set_value("obj");

		std::vector<T>* vec = ((std::vector<T>*)_Ptr(objOwner));
		for (auto& item : *vec) // item is T, which has __gobj
		{
			// Each item T will create its own node like <ItemGObjName ...>...</ItemGObjName>
			// So the vector_node will have children named after T's GObj name.
			item.__gobj.SaveXml(vector_node);
		}
	}
	virtual BOOL LoadXml(void *objOwner, const pugi::xml_node &node) override
	{
		if (strcmp(node.name(), elemname.c_str()) != 0) return FALSE;

		std::vector<T>* vec = ((std::vector<T>*)_Ptr(objOwner));
		vec->clear();
		
		// We need to know the GObj name of T to find the children
		T temp_item_for_name; // Assuming T is default constructible
		const char* item_gobj_name = temp_item_for_name.__gobj.GetName();

		for (pugi::xml_node item_obj_node = node.child(item_gobj_name); item_obj_node; item_obj_node = item_obj_node.next_sibling(item_gobj_name))
		{
			T new_item; // Create a new T
			if (new_item.__gobj.LoadXml(item_obj_node))
			{
				vec->push_back(new_item);
			}
			else
			{
				return FALSE; // Failed to load one of the objects in vector
			}
		}
		return TRUE;
	}

BOOL GLoadXml(const pugi::xml_document &doc) /*Entry point for loading GObj from XML doc*/\
{                                                                                   \
    pugi::xml_node root_node = doc.first_child();                                   \
    if (!root_node) return FALSE;                                                   \
    /* If SaveXml(doc) in GSaveXml makes the GObj's name the root, then find that: */ \
    /* pugi::xml_node gobj_node = doc.child(__gobj.GetName()); */                   \
    /* However, GObj<T>::SaveXml is designed to append to a parent_node. */        \
    /* So, the root node of the document should be the GObj itself. */             \
    return __gobj.LoadXml(root_node);                                             \
}                                                                                   \
